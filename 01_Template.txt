DEBUG = 0
if __name__ != "__main__":
    DEBUG = 1
    import DebugLib as dl

def MyInput():
    global DEBUG
    if DEBUG: return dl.DebugInput()
    else: return input()

def MyPrint(*arg, **keys):
    global DEBUG
    if DEBUG: return dl.DebugPrint(*arg, **keys)
    else: return print(*arg, **keys)
